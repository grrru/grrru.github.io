변수 확장

1${VAR:-word} # 기본값 할당: VAR가 비어있으면 word 사용
2${VAR:=word} # 기본값 대입: VAR가 비어있으면 word를 아예 저장함
3${VAR:?word} # 에러 출력: VAR가 비어있으면 word를 출력하고 종료
4${VAR:+word} # 대체값 사용: VAR가 있을 때만 word 사용
 1add_path_bash() {
 2  local mode="append"
 3
 4  case "${1-}" in
 5    --prepend|-p) mode="prepend"; shift ;;
 6    --append|-a)  mode="append"; shift ;;
 7  esac
 8
 9  [[ -z "${1-}" ]] && return 0
10  local p="$1"
11  [[ -d "$p" ]] || return 0
12
13  case "::$PATH::" in
14    *::"$p"::*) return 0 ;;
15  esac
16
17  if [[ "$mode" == "prepend" ]]; then
18    export PATH="$p${PATH:+:$PATH}"
19  else
20    export PATH="${PATH:+$PATH:}$p"
21  fi
22}
