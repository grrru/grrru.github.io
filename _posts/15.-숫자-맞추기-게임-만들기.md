```go
package main

import (
	"bufio"
	"fmt"
	"math/rand"
	"os"
	"time"
)

func main() {
	r := rand.New(rand.NewSource(time.Now().UnixNano()))
	reader := bufio.NewReader(os.Stdin)

	answer := r.Intn(1000)

	var t int

	for {
		fmt.Print("답을 입력하세요 : ")
		_, err := fmt.Scanln(&t)

		if err != nil {
			fmt.Println("숫자만 입력 가능합니다.")
			reader.ReadString('\n')
			continue
		}

		if t == answer {
			fmt.Printf("%d 정답~~\n", t)
			break
		}

		if t < answer {
			fmt.Printf("%d는 정답보다 작습니다.\n", t)
		} else {
			fmt.Printf("%d는 정답보다 큽니다.\n", t)
		}
	}
}
```
